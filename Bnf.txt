<BLOCK> ::= <IF> | <WHILE> | <ASSIGNMENT> | <FUNCCALL> | <EXIT>

<EXPRESSION> ::= <TERM> [<ADDOP> <TERM> ]*

<TERM> ::= <FACTOR> [<MULOP> <FACTOR>]*

<FACTOR> ::= ( <BOOLEXPRESSION> ) | <NUMBER> | <STRING> | <FUNCCALL>

<BOOLEXPRESSION> ::= <BOOLTERM> [<BOOLOR> <BOOLTERM>]*

<BOOLTERM> ::= <NOTFACTOR> [AND <NOTFACTOR>]*
<NOTFACTOR> ::= [NOT] <RELATION>
<RELATION> ::= <EXPRESSION> [<RELOP> 

---

GetNextToken ::= <'> <STRING-CONSTANT> <'> | <IDENTIFIER> | <NUMBER>

---

<PROGRAMMA> ::= <PROGRAM> [<VAR-DECL> <FUNC-DECL>]* <FUNC-MAIN> <END>

<VAR-DECL> ::= <TYPE> <IDENTIFIER>

<FUNC-DECL> ::= <TYPE> <IDENTIFIER> '(' [<PARAM-LIST] ')'

<FUNC-MAIN> ::= void main()

<PARAM-LIST> ::= [<VAR-DECL>]*


<ASSIGNMENT> ::= <VAR-NAME> = <BOOL-EXPRESSION>
<FUNC-CALL> ::= <VAR-NAME>.<FUNC-NAME>
